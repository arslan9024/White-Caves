# **Unified AI Assistant Dashboard: Company Owner Command Center**

## **Project: AI Assistant Orchestrator & Dashboard Hub**

### **Objective:** Create a unified React/Redux dashboard where the Company Owner can manage and access all AI assistants through an intelligent dropdown interface with real-time data visualization.

---

### **1.2 Redux State Management Structure**

**Redux Store Configuration:**
```javascript
// AI Assistant Dashboard Redux Slice
const aiAssistantDashboardSlice = createSlice({
  name: 'aiAssistantDashboard',
  initialState: {
    // Assistant Registry
    allAssistants: {
      byId: {},
      allIds: [],
      isLoading: false,
      lastFetched: null
    },
    
    // UI State
    ui: {
      selectedAssistant: 'mary',
      viewMode: 'dashboard', // 'dashboard', 'details', 'compare'
      layout: 'grid', // 'grid', 'list', 'focus'
      filters: {
        department: 'all',
        status: 'all',
        searchQuery: ''
      }
    },
    
    // Performance Data
    assistantPerformance: {
      // Aggregated metrics from all assistants
      overallHealth: 95,
      activeTasks: 0,
      criticalAlerts: [],
      recentActivity: []
    },
    
    // Quick Access
    favorites: [],
    recent: [],
    
    // Real-time Updates
    liveUpdates: {
      lastUpdate: null,
      connections: {}
    }
  },
  reducers: {
    // Action reducers
    selectAssistant: (state, action) => {
      state.ui.selectedAssistant = action.payload;
      // Add to recent
      if (!state.recent.includes(action.payload)) {
        state.recent.unshift(action.payload);
        if (state.recent.length > 5) state.recent.pop();
      }
    },
    toggleFavorite: (state, action) => {
      const index = state.favorites.indexOf(action.payload);
      if (index === -1) {
        state.favorites.push(action.payload);
      } else {
        state.favorites.splice(index, 1);
      }
    },
    updateAssistantMetrics: (state, action) => {
      // Update specific assistant metrics
      const { assistantId, metrics } = action.payload;
      if (state.allAssistants.byId[assistantId]) {
        state.allAssistants.byId[assistantId].metrics = {
          ...state.allAssistants.byId[assistantId].metrics,
          ...metrics
        };
      }
    }
  }
});
```

## **Part 2: Intelligent Assistant Dropdown Component**

### **2.1 Main Dropdown Component**

**AIAssistantSelector.jsx:**
```javascript
import React, { useState, useEffect } from 'react';
import { useSelector, useDispatch } from 'react-redux';
import { 
  selectAssistant, 
  toggleFavorite,
  updateAssistantMetrics 
} from '../redux/aiAssistantDashboardSlice';
import { 
  Search, 
  Star, 
  Activity, 
  TrendingUp, 
  Users,
  AlertCircle,
  Shield,
  DollarSign,
  Megaphone,
  MessageSquare,
  Briefcase,
  FileText
} from 'lucide-react';

const AIAssistantSelector = () => {
  const dispatch = useDispatch();
  const { allAssistants, ui, favorites, recent } = useSelector(
    state => state.aiAssistantDashboard
  );
  const [isOpen, setIsOpen] = useState(false);
  const [searchTerm, setSearchTerm] = useState('');
  
  // Assistant icons mapping
  const assistantIcons = {
    mary: <FileText className="w-5 h-5" />,
    theodora: <DollarSign className="w-5 h-5" />,
    olivia: <Megaphone className="w-5 h-5" />,
    zoe: <Briefcase className="w-5 h-5" />,
    laila: <Shield className="w-5 h-5" />,
    linda: <MessageSquare className="w-5 h-5" />,
    sophia: <Users className="w-5 h-5" />,
    daisy: <Users className="w-5 h-5" />
  };
  
  // Filter assistants based on search
  const filteredAssistants = Object.values(allAssistants.byId).filter(assistant => {
    if (!searchTerm) return true;
    return assistant.name.toLowerCase().includes(searchTerm.toLowerCase()) ||
           assistant.title.toLowerCase().includes(searchTerm.toLowerCase()) ||
           assistant.department.toLowerCase().includes(searchTerm.toLowerCase());
  });
  
  const currentAssistant = allAssistants.byId[ui.selectedAssistant];
  
  return (
    <div className="ai-assistant-selector">
      {/* Current Assistant Display */}
      <div 
        className="current-assistant-display"
        onClick={() => setIsOpen(!isOpen)}
      >
        <div className="assistant-avatar">
          <div className={`avatar-icon ${currentAssistant?.department}`}>
            {assistantIcons[currentAssistant?.id]}
          </div>
          <div className="avatar-status" 
               style={{ backgroundColor: currentAssistant?.metrics.systemHealth === 'optimal' ? '#10B981' : '#EF4444' }}
          />
        </div>
        <div className="assistant-info">
          <div className="assistant-name">{currentAssistant?.name}</div>
          <div className="assistant-title">{currentAssistant?.title}</div>
        </div>
        <div className="dropdown-arrow">
          {isOpen ? '▲' : '▼'}
        </div>
      </div>
      
      {/* Dropdown Menu */}
      {isOpen && (
        <div className="assistant-dropdown-menu">
          {/* Search Bar */}
          <div className="dropdown-search">
            <Search className="search-icon" />
            <input
              type="text"
              placeholder="Search AI assistants..."
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
              className="search-input"
              autoFocus
            />
          </div>
          
          {/* Favorites Section */}
          {favorites.length > 0 && (
            <div className="dropdown-section">
              <div className="section-header">
                <Star className="section-icon" />
                <span>Favorites</span>
              </div>
              {favorites.map(assistantId => {
                const assistant = allAssistants.byId[assistantId];
                if (!assistant) return null;
                return (
                  <AssistantItem
                    key={assistant.id}
                    assistant={assistant}
                    isFavorite={true}
                    onSelect={() => {
                      dispatch(selectAssistant(assistant.id));
                      setIsOpen(false);
                    }}
                    onToggleFavorite={() => dispatch(toggleFavorite(assistant.id))}
                  />
                );
              })}
            </div>
          )}
          
          {/* Recent Section */}
          {recent.length > 0 && (
            <div className="dropdown-section">
              <div className="section-header">
                <Activity className="section-icon" />
                <span>Recently Used</span>
              </div>
              {recent.map(assistantId => {
                const assistant = allAssistants.byId[assistantId];
                if (!assistant || favorites.includes(assistantId)) return null;
                return (
                  <AssistantItem
                    key={assistant.id}
                    assistant={assistant}
                    onSelect={() => {
                      dispatch(selectAssistant(assistant.id));
                      setIsOpen(false);
                    }}
                    onToggleFavorite={() => dispatch(toggleFavorite(assistant.id))}
                  />
                );
              })}
            </div>
          )}
          
          {/* All Assistants Section */}
          <div className="dropdown-section">
            <div className="section-header">
              <Users className="section-icon" />
              <span>All AI Assistants</span>
            </div>
            {filteredAssistants.map(assistant => {
              if (favorites.includes(assistant.id) || recent.includes(assistant.id)) return null;
              return (
                <AssistantItem
                  key={assistant.id}
                  assistant={assistant}
                  onSelect={() => {
                    dispatch(selectAssistant(assistant.id));
                    setIsOpen(false);
                  }}
                  onToggleFavorite={() => dispatch(toggleFavorite(assistant.id))}
                />
              );
            })}
          </div>
          
          {/* Quick Stats Footer */}
          <div className="dropdown-footer">
            <div className="quick-stats">
              <div className="stat">
                <TrendingUp className="stat-icon" />
                <span>{filteredAssistants.length} Assistants</span>
              </div>
              <div className="stat">
                <Activity className="stat-icon" />
                <span>{filteredAssistants.filter(a => a.metrics.systemHealth === 'optimal').length} Active</span>
              </div>
              {assistantPerformance.criticalAlerts.length > 0 && (
                <div className="stat alert">
                  <AlertCircle className="stat-icon" />
                  <span>{assistantPerformance.criticalAlerts.length} Alerts</span>
                </div>
              )}
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

// Assistant Item Component
const AssistantItem = ({ assistant, isFavorite = false, onSelect, onToggleFavorite }) => {
  const dispatch = useDispatch();
  
  return (
    <div 
      className="assistant-item"
      onClick={onSelect}
    >
      <div className="item-left">
        <div className={`item-avatar ${assistant.department}`}>
          {assistantIcons[assistant.id]}
        </div>
        <div className="item-info">
          <div className="item-name">{assistant.name}</div>

          <div className="item-metrics">

            <span className={`health-badge ${assistant.metrics.systemHealth}`}>

            </span>

          }}

            <div className="stat-value">{assistant.quickStats.today.value}</div>
            <div className="stat-label">{assistant.quickStats.today.label}</div>
          </div>
        )}

  );
};
```

```javascript
import React from 'react';

const AssistantPerformanceOverview = () => {
  const { allAssistants, assistantPerformance } = useSelector(
    state => state.aiAssistantDashboard
  );
  
  // Prepare data for charts
  const departmentData = Object.values(allAssistants.byId).reduce((acc, assistant) => {

            </BarChart>

        </div>
      </div>
      
      {/* Quick Stats */}
      <div className="quick-stats-grid">

const renderCustomizedLabel = ({ cx, cy, midAngle, innerRadius, outerRadius, percent, index }) => {
  const RADIAN = Math.PI / 180;
  const radius = innerRadius + (outerRadius - innerRadius) * 0.5;
  const x = cx + radius * Math.cos(-midAngle * RADIAN);
  const y = cy + radius * Math.sin(-midAngle * RADIAN);
  
  return (
    <text x={x} y={y} fill="white" textAnchor={x > cx ? 'start' : 'end'} dominantBaseline="central">
      {`${(percent * 100).toFixed(0)}%`}
    </text>
  );
};
```

## **Part 5: Real-time Updates & WebSocket Integration**

### **5.1 WebSocket Service for Live Updates**

**WebSocketService.js:**
```javascript
class AIAssistantWebSocketService {
  constructor(store) {
    this.store = store;
    this.socket = null;
    this.reconnectAttempts = 0;
    this.maxReconnectAttempts = 5;
  }
  
  connect() {
    const token = localStorage.getItem('auth_token');
    this.socket = new WebSocket(`wss://your-api/ws/ai-assistants?token=${token}`);
    
    this.socket.onopen = () => {
      console.log('AI Assistant WebSocket connected');
      this.reconnectAttempts = 0;
      this.store.dispatch({ type: 'websocket/CONNECTED' });
    };
    
    this.socket.onmessage = (event) => {
      const data = JSON.parse(event.data);
      this.handleMessage(data);
    };
    
    this.socket.onclose = () => {
      console.log('AI Assistant WebSocket disconnected');
      this.store.dispatch({ type: 'websocket/DISCONNECTED' });
      this.attemptReconnect();
    };
    
    this.socket.onerror = (error) => {
      console.error('WebSocket error:', error);
    };
  }
  
  handleMessage(data) {
    switch (data.type) {
      case 'assistant_metrics_update':
        this.store.dispatch(updateAssistantMetrics({
          assistantId: data.assistantId,
          metrics: data.metrics
        }));
        break;
        
      case 'assistant_health_change':
        this.store.dispatch({
          type: 'aiAssistantDashboard/updateAssistantHealth',
          payload: {
            assistantId: data.assistantId,
            health: data.health
          }
        });
        break;
        
      case 'new_alert':
        this.store.dispatch({
          type: 'aiAssistantDashboard/addAlert',
          payload: data.alert
        });
        break;
        
      case 'task_completed':
        this.store.dispatch({
          type: 'aiAssistantDashboard/updateTaskStatus',
          payload: data.task
        });
        break;
    }
  }
  
  attemptReconnect() {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      setTimeout(() => {
        console.log(`Attempting to reconnect (${this.reconnectAttempts}/${this.maxReconnectAttempts})`);
        this.connect();
      }, 3000 * this.reconnectAttempts);
    }
  }
  
  disconnect() {
    if (this.socket) {
      this.socket.close();
    }
  }
  
  sendMessage(type, payload) {
    if (this.socket && this.socket.readyState === WebSocket.OPEN) {
      this.socket.send(JSON.stringify({ type, ...payload }));
    }
  }
}

export default AIAssistantWebSocketService;
```
