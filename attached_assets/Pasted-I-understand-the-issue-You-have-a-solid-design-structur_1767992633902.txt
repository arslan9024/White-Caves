### **3. Create a Data Service (`dataService.js`)**

```javascript
// src/services/dataService.js

// Import all JSON files (Webpack will handle this)
import overviewData from '../data/dashboard/overview.json';
import leadsData from '../data/sales/leads.json';
import campaignsData from '../data/marketing/campaigns.json';
// ... import other JSON files

// Data service object that provides data to components
const dataService = {
  // Dashboard data
  getDashboardOverview: () => overviewData,
  
  // Sales data
  getLeads: (filter = {}) => {
    let data = leadsData;
    // Apply filters if provided
    if (filter.status) {
      data = data.filter(lead => lead.status === filter.status);
    }
    return data;
  },
  
  getLeadById: (id) => leadsData.find(lead => lead.id === id),
  
  // Marketing data
  getCampaigns: () => campaignsData,
  
  // Generic data fetching with caching
  cache: new Map(),
  
  fetchData: async (dataKey, queryParams = {}) => {
    const cacheKey = `${dataKey}-${JSON.stringify(queryParams)}`;
    
    if (this.cache.has(cacheKey)) {
      return this.cache.get(cacheKey);
    }
    
    // Simulate API call delay
    await new Promise(resolve => setTimeout(resolve, 100));
    
    let data;
    switch (dataKey) {
      case 'overview':
        data = overviewData;
        break;
      case 'leads':
        data = this.getLeads(queryParams);
        break;
      case 'campaigns':
        data = campaignsData;
        break;
      default:
        data = [];
    }
    
    this.cache.set(cacheKey, data);
    return data;
  }
};

export default dataService;
```

## **PHASE 3: Create Content Components**

### **4. Create Dashboard Content Component**

```jsx
// src/views/DashboardView.jsx
import React, { useState, useEffect } from 'react';
import {
  Grid,
  Paper,
  Typography,
  CircularProgress,
  Alert,
} from '@mui/material';
import { 
  TrendingUp,
  People,
  ShoppingCart,
  AttachMoney 
} from '@mui/icons-material';
import dataService from '../services/dataService';
import MetricCard from '../components/shared/MetricCard';

const DashboardView = () => {
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  const [dashboardData, setDashboardData] = useState(null);
  const [metrics, setMetrics] = useState([]);

  useEffect(() => {
    const loadData = async () => {
      try {
        setLoading(true);
        
        // Fetch dashboard data from JSON
        const [overview, leads] = await Promise.all([
          dataService.fetchData('overview'),
          dataService.fetchData('leads', { status: 'new' })
        ]);
        
        // Process data for display
        const processedMetrics = [
          {
            title: 'Total Revenue',
            value: overview?.revenue || 0,
            icon: <AttachMoney />,
            color: 'success',
            trend: overview?.revenueTrend || 'up',
            change: '12%'
          },
          {
            title: 'New Leads',
            value: leads?.length || 0,
            icon: <People />,
            color: 'info',
            trend: 'up',
            change: '24%'
          },
          {
            title: 'Conversion Rate',
            value: `${overview?.conversionRate || 0}%`,
            icon: <TrendingUp />,
            color: 'warning',
            trend: 'stable',
            change: '2%'
          },
          {
            title: 'Pending Orders',
            value: overview?.pendingOrders || 0,
            icon: <ShoppingCart />,
            color: 'primary',
            trend: 'down',
            change: '-5%'
          }
        ];
        
        setDashboardData(overview);
        setMetrics(processedMetrics);
        setLoading(false);
      } catch (err) {
        setError('Failed to load dashboard data');
        setLoading(false);
      }
    };
    
    loadData();
  }, []);

  if (loading) {
    return (
      <Box display="flex" justifyContent="center" alignItems="center" height="400px">
        <CircularProgress />
      </Box>
    );
  }

  if (error) {
    return <Alert severity="error">{error}</Alert>;
  }

  return (
    <Box>
      {/* Page Header */}
      <Box mb={4}>
        <Typography variant="h4" fontWeight="bold">
          Dashboard Overview
        </Typography>
        <Typography variant="body1" color="text.secondary">
          Welcome back! Here's what's happening today.
        </Typography>
      </Box>

      {/* Metrics Grid */}
      <Grid container spacing={3} mb={4}>
        {metrics.map((metric, index) => (
          <Grid item xs={12} sm={6} md={3} key={index}>
            <MetricCard {...metric} />
          </Grid>
        ))}
      </Grid>

      {/* Additional Content Sections */}
      <Grid container spacing={3}>
        {/* Recent Activity */}
        <Grid item xs={12} md={8}>
          <Paper sx={{ p: 3, height: '100%' }}>
            <Typography variant="h6" gutterBottom>
              Recent Activity
            </Typography>
            {/* Add activity list component here */}
          </Paper>
        </Grid>

        {/* Quick Stats */}
        <Grid item xs={12} md={4}>
          <Paper sx={{ p: 3, height: '100%' }}>
            <Typography variant="h6" gutterBottom>
              Quick Stats
            </Typography>
            {/* Add stats component here */}
          </Paper>
        </Grid>
      </Grid>
    </Box>
  );
};

export default DashboardView;
```
