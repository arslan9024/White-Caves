import React, { useState } from 'react';
import './PropertyCard.css'; // Preserve all existing CSS classes
import './PropertyCard.upgraded.css'; // ADD NEW STYLES ONLY

const PropertyCard = ({ property, onFavorite, onViewDetails }) => {
  // PRESERVE ALL EXISTING LOGIC
  const [isFavorite, setIsFavorite] = useState(property.isFavorite);
  const [imageLoaded, setImageLoaded] = useState(false);
  
  // PRESERVE ALL EXISTING FUNCTIONS
  const handleFavoriteClick = () => {
    setIsFavorite(!isFavorite);
    onFavorite(property.id, !isFavorite);
  };
  
  const handleViewDetails = () => {
    onViewDetails(property.id);
  };
  
  // ONLY UPDATE RETURN JSX WITH NEW STYLING CLASSES
  return (
    <div className="property-card upgraded-card">
      {/* Preserve all existing div structure */}
      <div className="property-image-container">
        <img 
          src={property.imageUrl} 
          alt={property.title}
          className={`property-image ${imageLoaded ? 'loaded' : 'loading'}`}
          onLoad={() => setImageLoaded(true)}
        />
        <div className="property-badge upgraded-badge">
          {property.type}
        </div>
        <button 
          className={`favorite-btn ${isFavorite ? 'active' : ''} upgraded-favorite`}
          onClick={handleFavoriteClick}
          aria-label={isFavorite ? 'Remove from favorites' : 'Add to favorites'}
        >
          â™¥
        </button>
      </div>
      
      <div className="property-info upgraded-info">
        <h3 className="property-title upgraded-title">{property.title}</h3>
        <p className="property-address upgraded-address">{property.address}</p>
        
        <div className="property-features upgraded-features">
          <span className="feature">
            <i className="icon-bed"></i> {property.bedrooms}
          </span>
          <span className="feature">
            <i className="icon-bath"></i> {property.bathrooms}
          </span>
          <span className="feature">
            <i className="icon-size"></i> {property.size} sqft
          </span>
        </div>
        
        <div className="property-price upgraded-price">
          <span className="price-amount">${property.price.toLocaleString()}</span>
          {property.pricePerSqft && (
            <span className="price-unit">${property.pricePerSqft}/sqft</span>
          )}
        </div>
        
        <button 
          className="view-details-btn upgraded-btn"
          onClick={handleViewDetails}
        >
          View Details
        </button>
      </div>
    </div>
  );
};

// EXPORT EXACTLY AS BEFORE
export default PropertyCard;
