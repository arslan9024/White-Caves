const properties = require('../../data/properties.json');

class MockPropertyService {
  async getAll(filters = {}) {
    // Filter logic on JSON data
    return properties.data.filter(p => this.matchesFilters(p, filters));
  }
  
  async getById(id) {
    return properties.data.find(p => p.property_id === id);
  }
  
  async create(propertyData) {
    const newProperty = {
      property_id: this.generateId(),
      ...propertyData,
      created_at: new Date().toISOString()
    };
    properties.data.push(newProperty);
    this.saveToFile();
    return newProperty;
  }
}
