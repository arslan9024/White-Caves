// Export as CSS custom properties for global use
export function createGlobalStyles() {
  const styleSheet = document.createElement('style');
  let cssVariables = ':root {\n';
  
  // Convert tokens to CSS custom properties
  function flattenTokens(obj, prefix = '') {
    for (const [key, value] of Object.entries(obj)) {
      const newPrefix = prefix ? `${prefix}-${key}` : `--${key}`;
      
      if (typeof value === 'object' && !Array.isArray(value)) {
        flattenTokens(value, newPrefix);
      } else {
        cssVariables += `  ${newPrefix}: ${value};\n`;
      }
    }
  }
  
  flattenTokens(tokens);
  cssVariables += '}';
  styleSheet.textContent = cssVariables;
  
  return styleSheet;
}

// Helper function to use in styled-components or emotion
export function token(path) {
  const keys = path.split('.');
  let result = tokens;
  
  for (const key of keys) {
    if (result[key] === undefined) {
      console.warn(`Design token "${path}" not found`);
      return '';
    }
    result = result[key];
  }
  
  return result;
}
