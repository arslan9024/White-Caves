```typescript
// types/property.types.ts
export interface PropertyLocation {
  community: string;
  tower?: string;
  district?: string;
  emirate: string;
}

export interface PropertyMedia {
  primary_image: string;
  gallery: string[];
  virtual_tour_url?: string;
  floor_plan_url?: string;
}

export interface PropertyFeature {
  name: string;
  icon?: string; // Optional icon class for visual representation
}

export interface DailyProperty {
  property_ref: string;
  title: string;
  price_aed: number;
  location: PropertyLocation;
  media: PropertyMedia;
  features: PropertyFeature[];
  property_type: string; // 'Apartment', 'Villa', 'Penthouse', etc.
  bedrooms: number;
  bathrooms: number;
  area_sqft: number;
  status: string;
  description?: string;
  created_date?: string;
}

export interface FeaturedPropertiesResponse {
  success: boolean;
  date_active: string;
  featured_properties: string[];
  property_details: DailyProperty[];
  generated_at: string;
  is_fallback?: boolean;
  source?: string;
}
```

### **2. Main Property Display Component**
Here's the core component that fetches and displays the properties:

```tsx
// components/PropertyDisplay/PropertyDisplay.tsx
import React, { useState, useEffect } from 'react';
import axios from 'axios';
import { FeaturedPropertiesResponse, DailyProperty } from '../../types/property.types';
import './PropertyDisplay.css';
import PropertyCard from './PropertyCard';
import LoadingSkeleton from './LoadingSkeleton';

// Dubai-specific emojis/icons mapping
const PROPERTY_ICONS = {
  'Apartment': 'üè¢',
  'Villa': 'üè°', 
  'Penthouse': 'üëë',
  'Townhouse': 'üèòÔ∏è',
  'Duplex': 'üè†',
  'Office': 'üíº',
  'Retail': 'üõçÔ∏è'
};

const PropertyDisplay: React.FC = () => {
  const [properties, setProperties] = useState<DailyProperty[]>([]);
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<string | null>(null);
  const [activeDate, setActiveDate] = useState<string>('');
  const [viewMode, setViewMode] = useState<'grid' | 'list'>('grid');
  const [selectedProperty, setSelectedProperty] = useState<DailyProperty | null>(null);
  const [currentImageIndex, setCurrentImageIndex] = useState<number>(0);

  // API endpoint - configure based on your environment
  const API_ENDPOINT = process.env.REACT_APP_API_URL || '/api/featured-properties';

  useEffect(() => {
    fetchFeaturedProperties();
    
    // Optional: Auto-refresh every hour
    const refreshInterval = setInterval(() => {
      fetchFeaturedProperties();
    }, 60 * 60 * 1000); // 1 hour
    
    return () => clearInterval(refreshInterval);
  }, []);

  const fetchFeaturedProperties = async () => {
    try {
      setLoading(true);
      const response = await axios.get<FeaturedPropertiesResponse>(API_ENDPOINT);
      
      if (response.data.success && response.data.property_details) {
        setProperties(response.data.property_details);
        setActiveDate(new Date(response.data.date_active).toLocaleDateString('en-AE', {
          weekday: 'long',
          year: 'numeric',
          month: 'long',
          day: 'numeric',
          timeZone: 'Asia/Dubai'
        }));
      } else {
        throw new Error('No properties available');
      }
    } catch (err) {
      setError('Unable to load featured properties. Please try again later.');
      console.error('Error fetching properties:', err);
    } finally {
      setLoading(false);
    }
  };

  const formatPrice = (price: number): string => {
    if (price >= 1000000) {
      return `AED ${(price / 1000000).toFixed(2)}M`;
    } else if (price >= 1000) {
      return `AED ${(price / 1000).toFixed(0)}K`;
    }
    return `AED ${price}`;
  };

  const openModal = (property: DailyProperty) => {
    setSelectedProperty(property);
    setCurrentImageIndex(0);
    document.body.style.overflow = 'hidden'; // Prevent background scrolling
  };

  const closeModal = () => {
    setSelectedProperty(null);
    document.body.style.overflow = 'auto';
  };

  const nextImage = () => {
    if (selectedProperty && selectedProperty.media.gallery.length > 0) {
      setCurrentImageIndex((prev) => 
        prev === selectedProperty.media.gallery.length - 1 ? 0 : prev + 1
      );
    }
  };

  const prevImage = () => {
    if (selectedProperty && selectedProperty.media.gallery.length > 0) {
      setCurrentImageIndex((prev) => 
        prev === 0 ? selectedProperty.media.gallery.length - 1 : prev - 1
      );
    }
  };

  if (loading) {
    return <LoadingSkeleton />;
  }

  if (error) {
    return (
      <div className="property-error">
        <div className="error-icon">‚ö†Ô∏è</div>
        <h3>Unable to Load Properties</h3>
        <p>{error}</p>
        <button onClick={fetchFeaturedProperties} className="retry-button">
          Try Again
        </button>
      </div>
    );
  }

  return (
    <div className="property-display-container">
      {/* Header Section */}
      <div className="property-header">
        <div className="header-content">
          <h1 className="section-title">
            <span className="highlight">Featured Properties</span> in Dubai
          </h1>
          <p className="section-subtitle">
            Curated by Olivia AI ‚Ä¢ Updated Daily ‚Ä¢ {activeDate}
          </p>
          <div className="property-stats">
            <span className="stat-badge">
              <span className="stat-icon">üéØ</span>
              {properties.length} Handpicked
            </span>
            <span className="stat-badge">
              <span className="stat-icon">üîÑ</span>
              Daily Rotation
            </span>
            <span className="stat-badge">
              <span className="stat-icon">ü§ñ</span>
              AI-Powered Selection
            </span>
          </div>
        </div>
        
        <div className="view-controls">
          <button 
            onClick={() => setViewMode('grid')} 
            className={`view-button ${viewMode === 'grid' ? 'active' : ''}`}
            aria-label="Grid view"
          >
            ‚ñ¶ Grid
          </button>
          <button 
            onClick={() => setViewMode('list')} 
            className={`view-button ${viewMode === 'list' ? 'active' : ''}`}
            aria-label="List view"
          >
            ‚â° List
          </button>
        </div>
      </div>

      {/* Properties Grid/List */}
      <div className={`properties-container ${viewMode}-view`}>
        {properties.map((property) => (
          <PropertyCard
            key={property.property_ref}
            property={property}
            viewMode={viewMode}
            formatPrice={formatPrice}
            onViewDetails={() => openModal(property)}
          />
        ))}
      </div>

      {/* Property Detail Modal */}
      {selectedProperty && (
        <div className="property-modal" onClick={closeModal}>
          <div className="modal-content" onClick={(e) => e.stopPropagation()}>
            {/* Modal Header */}
            <div className="modal-header">
              <h2>{selectedProperty.title}</h2>
              <button className="close-modal" onClick={closeModal}>√ó</button>
            </div>

            {/* Image Gallery */}
            <div className="modal-gallery">
              <div className="main-image-container">
                <img 
                  src={selectedProperty.media.gallery[currentImageIndex] || selectedProperty.media.primary_image} 
                  alt={selectedProperty.title}
                  className="main-image"
                />
                <button className="nav-button prev" onClick={prevImage}>&lt;</button>
                <button className="nav-button next" onClick={nextImage}>&gt;</button>
                <div className="image-counter">
                  {currentImageIndex + 1} / {selectedProperty.media.gallery.length || 1}
                </div>
              </div>
              
              {selectedProperty.media.gallery.length > 1 && (
                <div className="thumbnail-grid">
                  {selectedProperty.media.gallery.map((img, idx) => (
                    <img
                      key={idx}
                      src={img}
                      alt={`View ${idx + 1}`}
                      className={`thumbnail ${currentImageIndex === idx ? 'active' : ''}`}
                      onClick={() => setCurrentImageIndex(idx)}
                    />
                  ))}
                </div>
              )}
            </div>

            {/* Property Details */}
            <div className="modal-details">
              <div className="price-section">
                <h3 className="modal-price">{formatPrice(selectedProperty.price_aed)}</h3>
                <span className="property-type">
                  {PROPERTY_ICONS[selectedProperty.property_type as keyof typeof PROPERTY_ICONS] || 'üè†'} 
                  {selectedProperty.property_type}
                </span>
              </div>

              <div className="specs-grid">
                <div className="spec-item">
                  <span className="spec-icon">üõèÔ∏è</span>
                  <span className="spec-value">{selectedProperty.bedrooms}</span>
                  <span className="spec-label">Bedrooms</span>
                </div>
                <div className="spec-item">
                  <span className="spec-icon">üöø</span>
                  <span className="spec-value">{selectedProperty.bathrooms}</span>
                  <span className="spec-label">Bathrooms</span>
                </div>
                <div className="spec-item">
                  <span className="spec-icon">üìê</span>
                  <span className="spec-value">{selectedProperty.area_sqft.toLocaleString()} sqft</span>
                  <span className="spec-label">Area</span>
                </div>
                <div className="spec-item">
                  <span className="spec-icon">üìç</span>
                  <span className="spec-value">{selectedProperty.location.community}</span>
                  <span className="spec-label">Community</span>
                </div>
              </div>

              {/* Features */}
              <div className="features-section">
                <h4>Key Features</h4>
                <div className="features-list">
                  {selectedProperty.features.map((feature, idx) => (
                    <span key={idx} className="feature-tag">
                      {feature.icon} {feature.name}
                    </span>
                  ))}
                </div>
              </div>

              {/* Description */}
              {selectedProperty.description && (
                <div className="description-section">
                  <h4>Description</h4>
                  <p>{selectedProperty.description}</p>
                </div>
              )}

              {/* Action Buttons */}
              <div className="modal-actions">
                <button className="primary-action">
                  üìû Schedule Viewing
                </button>
                <button className="secondary-action">
                  üìÑ Download Brochure
                </button>
                {selectedProperty.media.virtual_tour_url && (
                  <button className="virtual-tour-button">
                    üé• Virtual Tour
                  </button>
                )}
              </div>
            </div>
          </div>
        </div>
      )}

      {/* Footer Note */}
      <div className="ai-note">
        <p>
          <strong>ü§ñ AI-Powered Selection:</strong> These properties are automatically selected daily by Olivia, 
          our AI assistant, from White Caves' available inventory. 
          Properties refresh every 24 hours.
        </p>
      </div>
    </div>
  );
};

export default PropertyDisplay;
```
